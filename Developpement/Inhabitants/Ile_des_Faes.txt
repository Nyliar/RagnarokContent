	mes "Je veux bien t'envoyer de l'autre côté, si ton niveau dépasse 65 et que tu es conscient des riques.";
	switch(select("Laisse moi passer","Je reviendrais plus tard alors"))
	{
		case 1:
			if (BaseLevel > 65)

	end;
