
thalys,173,140,3	script	Mary l'aubergiste	701,{

	set .@npcname$, "[Mary]";
	
	mes .@npcname$;
	mes "";
	mes "Bonjour, " + strcharinfo(0);
	mes "Je tiens l'unique auberge de tout Thalys.";
	mes "Souhaites-tu y dormir?";
	mes "Nous offrons le gîte aux gens de faible niveau, jusqu'au niveau 30.";
	mes "Alors ? Intéressé ?";
	switch(select("Oui", "Non"))
	{
		case 1:
			next;
			mes "Ok, bonne nuit!";
			if (BaseLevel <= 30) goto dodo;
			else
			{
				mes "Nous prenons un tarif dépendant de ton niveau.";
				mes "Je vois que tu es niveau " + BaseLevel + ",";
				set .@ratio, 2;
				if (Class < 4001)
					set .@ratio, 1;
				set .@count, (BaseLevel/10 * .hprice * .@ratio) + (BaseLevel%10 * .bprice * .@ratio);
				mes "tu me dois donc " + .@count + " Zenys. Ok ?";
				switch(select("Oui", "Non"))
				{
					case 1:
						if (Zeny >= .@count)
						{
							set Zeny, Zeny - .@count;
							goto dodo;
						}
						else
						{
							mes "Désolé, mais tu n'as pas ce qu'il faut.";
							mes "Reviens avec " + .@count + " Zenys, et je te logerais.";
						}
						break;
					case 2:
						mes "Tu ne trouveras pas moins cher, tu sais ?!";
						break;
				}
			}			
			break;
		case 2:
			next;
			mes "Bien, tu connais la maison maintenant.";
			mes "A bientôt.";
			break;
	}
	close;

dodo:
	percentheal 100, 100;
	close;

OnInit:
	set .hprice, 500;
	set .bprice, 50;
	end;

L_Red_Price:
	set .hprice, 250;
	set .bprice, 25;
	return;

L_Up_Price:
	set .hprice, 500;
	set .bprice, 50;
	return;
}

				switch(select("Oui", "Non"))
				{
					case 1:
						if (Zeny >= .@count)
						{
							set zeny,zeny-.@count;
							goto buff;
						} 
						else
						{
							mes "Désolé, mais tu n'as pas ce qu'il faut.";
							mes "Reviens avec " + .@count + " Zenys, et je te bénierais.";

		case 2:
			mes "Les Dieux ne sont pas rancunier. Reviens me voir.";
			mes "A bientôt.";
			break;

	end;

	set .bprice, 150;
	return;
}
