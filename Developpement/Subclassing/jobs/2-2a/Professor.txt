valkyrie,53,47,3	script	Professeur#Valkyrie	743,{
	if (SUBCLASSING == 1 && Class != 4017 && Class != 4010)
	{
		mes "[Professeur]";
		mes "Tu as choisi le métier de Professor en tant que classe Vane?";
		mes "Es-tu bien sûr de ton choix? Il n'y a pas de machine arrière.";
		switch(select("Oui","Non","Je ne sais pas, tu peux m'en dire plus?"))
		{
			case 1:
				mes "Très bien. Qu'il en soit ainsi.";
				set SUBCLASSING, 4017; 
				getitem 21098, 1;
				break;
			case 2:
				mes "Tu as raison. Réfléchis bien.";
				break;
			case 3:
				mes "Tu veux en savoir plus? Bien sûr!";
				mes "C'est logique. Vu le poids de ta décision sur ton futur.";
				mes "En tant que Professor, tu peux accéder aux skills suivants:";
				mes "Health Conversion, Soul Change, Soul Burn, Mind Breaker, Memorize, Wall of Fog, Spider Web.";
				mes "Earth Spike, Heaven's Drive, Magic Rod, Spell Breaker, Auto Spell et Dispell.";
				mes "Voilà, tu en sais un peu plus. Réfléchis bien.";
				break;
		} 
		close;
	}
	else if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[Professeur]";
			mes "Félicitations.";
			mes "Honneur aux Guerriers!";
			close;
		}
		mes "[Professeur]";
		mes "Cela peut prendre une vie...";
		mes "Oui, une vie entière, pour amasser les connaissances nécessaires afin de devenir Professeur...";
		next;
		mes "[Professeur]";
		mes "C'est accablant.";
		mes "Plus vous apprenez, plus vous découvrez qu'il existe d'autres choses que vous ignorez.";
		mes "Il n'y a pas de fin au processus d'apprentissage...";
		close;
	}
	else if (ADVJOB == Job_Professor && Class == Job_Mage_High && JobLevel > 39) {
		mes "[Professeur]";
		mes "Rune-Midgard ne compte pas assez de Professeurs pour atteindre une ère de prospérité. Le monde a besoin de vous...";
		next;
		mes "[Professeur]";
		mes "Serez-vous prêt à endosser cette responsabilité?";
		mes "Souhaitez-vous servir Rune-Midgard en tant que Professeur?";
		next;
		if (select("Non.:Oui.") == 1) {
			mes "[Professeur]";
			mes "Revenez quand vous vous sentirez prêt.";
			next;
			mes "[Professeur]";
			mes "Honneur aux Guerriers!";
			close;
		}
		if (SkillPoint) {
			mes "[Professeur]";
			mes "Vous n'avez pas terminé votre apprentissage.";
			close;
		}
		jobchange Job_Professor;
		set ADVJOB,0;
		mes "[Professeur]";
		mes "Félicitations!";
		mes "En tant que Professeur, j'espère que vous vous appliquerez à mettre en lumière les connaissances restées obscures.";
		close;
	}

	mes "[Professeur]";
	mes "Bienvenue au Walhalla, le Hall des Honneurs.";
	next;
	mes "[Professeur]";
	mes "Faites comme chez vous.";
	mes "Honneur aux Guerriers!";
	close;
}
