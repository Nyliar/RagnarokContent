	set .@npcname$, "[Moine Sombre]";
	mes .@npcname$;
	mes "";
	mes "Bonjour à toi.";
	mes "Je vois que tu as déjà eu de nombreux choix, de nombreuses vies.";
	mes "Sais-tu que tout ce que tu as rejeté ne meurs pas vraiment.";
	if (FOLK_Q == 0)
	{
		next;
		mes .@npcname$;
		mes "Ces vies gâchées finissent par revivre en Fölkvang, la Plaine des Oubliés.";
		mes "C'est un endroit...froid. La mort y règne en maître.";
		set FOLK_Q, 1;
	}
	else if (FOLK_Q == 2)
	{
		if (countitem(7109) >= 1)
		{
			next;
			mes .@npcname$;
			mes "Bien. Tu sais donc à quoi tu t'engages.";
			mes "Soit. Libre à toi d'aller sur la Plaine.";
			delitem 7109, 1;
		}
		set FOLK_Q, 3;
	}
	close;
}
