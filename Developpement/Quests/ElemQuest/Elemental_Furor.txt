							set $element_rank, 2;

							set $element_rank, 3;

						donpcevent "yggshoo::OnBloomAwake";

						set .@map$, "shitsu";
						killmonster .@map$,"All";

							copyarray .@partyAid[0], $@partymemberaid[0], getarraysize($@partymemberaid);

								attachrid(.@partyAid);
