capitale,217,167,4	script	Pillar::cappill1	406,{
	donpcevent "Crash Pillar::OnNpcEnabled";
	end;

OnInit:
	hideonnpc strnpcinfo(3);
	end;

OnNPCKillEvent:
	if (killedrid == 2536)
	{
		hideoffnpc "cappill1";
		set $mimir_killed, 1;
		startnpctimer;
		donpcevent "alalynque::OnCheckStartEvent";
	}
	end;

OnTimer3600000:
	//one hour delay
	set $mimir_killed, 0;
	donpcevent "alalynque::OnCloseEvent";
	hideonnpc strnpcinfo(3);
	end;

OnAllMvpKilled:
	npcwalkto 222,172;
	hideonnpc "cappill1";
	donpcevent "crashpillar::OnNpcEnabled";
	end;
}
