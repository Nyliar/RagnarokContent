mysterio,83,86,5	script	Weird Statue#1::etrsta1	111,{
	if (#quetemysterio >= 1)
	{	
		if (#lastactivate == (atoi(strnpcinfo(2)) - 1))
		{
			set #correct, #correct+1;
			set #lastactivate, atoi(strnpcinfo(2));
		}
		else 
		{
			set #correct, 0;
			set #lastactivate, 0;
		}
		if(LANGUAGE == 2)
		{		
			mes "Me: " + atoi(strnpcinfo(2));
			mes "You: " + #correct;
		}
		else if (LANGUAGE == 1)
		{
			mes "Moi: " + atoi(strnpcinfo(2));
			mes "Toi: " + #correct;
		}

		if (#correct == 10)
		{
			next;
			if(LANGUAGE == 2)
			{
				mes "Congratulation. You've successfully fill the ordeal to access to the World Snake Lair.";
				set #quetemysterio, 3;
				mes "To enter in it, you'll now have to prove your valor.";
				mes "You'll have to fight and win against the Jormungand monsters.";
				mes "Get prepared.";
			}
			else if (LANGUAGE == 1)
			{
				mes "Félicitation. Vous avez accompli avec brio l'épreuve pour accéder à l'Antre du Serpent Monde.";
				set #quetemysterio, 3;
				mes "Mais pour entrer, vous devrez encore prouver votre valeur.";
				mes "Vous devrez vous battre et vaincre les monstres de Jormungand.";
				mes "Préparez-vous.";
			}
			killmonsterall "npc2";
			warp "npc2", 0, 0;
			setmapflag "npc2", "mf_noteleport";
			setmapflag "npc2", "mf_nowarp";
			areamonster "npc2",30,69,69,29,"Jormungand Henchman",1372,20,"etrsta1::OnSbireDead";
		}
	}
	else
	{
		if(LANGUAGE == 2)
		{
			mes "It a rock statue.";	
			mes "It looks weird, but you can't find anything suspicious.";
		}
		else if (LANGUAGE == 1)
		{
			mes "C'est une statue de pierre.";	
			mes "Elle a l'air étrange, mais vous ne trouvez rien de particulier.";
		}
	}
	close;

	set .@mob_dead_num,mobcount("npc2", "etrsta1::OnSbireDead");
	if (.@mob_dead_num < 1)
	{
		atcommand "!cleanmap";
		if(LANGUAGE == 2)
		{
			dispbottom "[Lair Guardian]: Congratulation. You can continue your accession to the Lair.";
			dispbottom "[Lair Guardian]: You'll now have to find me.";
		}
		else if (LANGUAGE == 1)
		{
			dispbottom "[Lair Guardian]: Félicitation, vous pouvez continuer votre ascension vers l'Antre.";
			dispbottom "[Lair Guardian]: Vous devrez maintenant me trouver.";
		}

		set #quetemysterio, 4;
		mapwarp "npc2", "mysterio", 0, 0;
	}
	else
	{
		if(LANGUAGE == 2)
		{
			dispbottom "[Lair Guardian]: There's still " + .@mob_dead_num + "!";
		}
		else if (LANGUAGE == 1)
		{
			dispbottom "[Lair Guardian]: Il en reste " + .@mob_dead_num + "!";
		}
	}
	end;

}
