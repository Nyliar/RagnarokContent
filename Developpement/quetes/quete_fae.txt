					set .diecount, .diecount+1;
					if (.diecount == 9)
						set .diecount, 1;
					close2;
					disablenpc "scint";
					
					do
					{		
							set .movex,rand(0,150);
							set .movey,rand(0,150);
							set .map$,strcharinfo(3);
					}
					while(!checkcell(.map$,.movex,.movey,cell_chkpass));
					movenpc strnpcinfo(3),.movex,.movey;				
					enablenpc "scint";
					close;
				}
			}
			else
			{
				mes "[Fairy]";
				mes "Au secours!";
				mes "Un homme mauvais m'a avalé un jour,";
				mes "et depuis, je ne peux plus voler.";
				mes "Il m'a volé toute ma poussière de Fae...";
				mes "Aide moi, et je t'aiderais en retour...";
				next;
				mes "Rapporte moi de la Poussière de Sorcière, pour relancer ma production,";
				mes "10";
				mes "Ensuite, des Fleurs d'Illusions, pour communiquer avec Freya,";
				mes "5";
				mes "Et enfin, de la Lumière, fournit par des Flashlight,";
				mes "30...";
				next;
				mes "Vite, ma lumière s'éteint...";
				mes "Je mourrais d'ici à 15 minutes...";
				addtimer 900000, strnpcinfo(3)+"::OnFaeDead";
				set encours, 1;
				set DROPXP_Q, 1;
			}
		}
	}		
	close;
	
OnInit:
	set .diecount, 0;
	set .faedead, 0;
	close;

OnFaeDead:
	set .faedead, 1;
	deltimer strnpcinfo(3)+"::OnFaeDead";
	close;
